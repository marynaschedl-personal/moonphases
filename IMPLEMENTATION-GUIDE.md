**Current**: Static gradient circle
**Approach**: Add CSS animations + subtle JavaScript updates

```javascript
// Add to your main JavaScript
function animateMoon(illumination) {
  const moon = document.getElementById('moonVisual');
  
  // Create animated shadow effect based on phase
  const shadowBlur = 30 - (illumination * 0.2);
  const shadowColor = `rgba(0, 0, 0, ${0.5 + illumination * 0.3})`;
  
  moon.style.transition = 'all 300ms ease-out';
  moon.style.boxShadow = `0 0 ${shadowBlur}px ${shadowColor}`;

}
```

```javascript
function initPhaseTimeline() {
  const timelineScroll = document.querySelector('.timeline-scroll');
  const timelineCards = document.querySelectorAll('.timeline-card');
  const todayCard = timelineScroll.querySelector('[data-today]');
  
  // Scroll to today on load
  if (todayCard) {
    timelineScroll.scrollLeft = todayCard.offsetLeft - 50;
  }
  
  // Add click handlers for card selection
  timelineCards.forEach(card => {
    card.addEventListener('click', function() {
      // Remove active from all
      timelineCards.forEach(c => c.classList.remove('active'));
      // Add active to clicked
      this.classList.add('active');
      
      // Show details for this date
      const date = this.dataset.date;
      showTimelineDetails(date);
      
      // Smooth scroll to center
      this.scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'center' });
    });
  });
}
```

## 3️⃣ Enhancement: Dynamic Background System

### Goal
Make the background respond to moon phase and current time for immersion.

### Implementation

#### A. Background Color Logic
```javascript
function updateBackgroundByPhase(illumination) {
  const container = document.querySelector('body');
  let bgColor, glowColor;
  
  // Determine colors based on illumination
  if (illumination < 10) {
    // New moon - deep dark
    bgColor = 'linear-gradient(135deg, #0B0B0F 0%, #1A1A2E 100%)';
    glowColor = 'rgba(0, 0, 0, 0)';
  } else if (illumination < 25) {
    // Crescent - dark with hint of glow
    bgColor = 'linear-gradient(135deg, #0F0F1F 0%, #1A2540 100%)';
    glowColor = 'rgba(255, 215, 0, 0.1)';
  } else if (illumination < 75) {
    // Quarter to gibbous - medium brightness
    bgColor = 'linear-gradient(135deg, #1A2A3A 0%, #2A4A5A 100%)';
    glowColor = 'rgba(255, 215, 0, 0.15)';
  } else {
    // Full moon - bright
    bgColor = 'linear-gradient(135deg, #3A5A6A 0%, #4A7A8A 100%)';
    glowColor = 'rgba(255, 215, 0, 0.3)';
  }
  
  container.style.background = bgColor;
  container.style.boxShadow = `inset 0 0 100px ${glowColor}`;
  container.style.transition = 'all 800ms ease-out';
}
```

```javascript
function createStarField(illumination) {
  const starField = document.getElementById('starField');
  starField.innerHTML = ''; // Clear existing
  
  // Fewer stars as moon brightens
  const starCount = Math.max(30, Math.floor(100 - (illumination * 0.8)));
  const starOpacity = Math.max(0.2, 1 - (illumination * 0.01));
  
  for (let i = 0; i < starCount; i++) {
    const star = document.createElement('div');
    star.className = 'star' + (Math.random() > 0.7 ? ' twinkling' : '');
    star.style.left = Math.random() * 100 + '%';
    star.style.top = Math.random() * 100 + '%';
    star.style.opacity = starOpacity * (0.5 + Math.random());
    starField.appendChild(star);
  }
}
```

```javascript
function toggleSection(button) {
  const content = button.nextElementSibling;
  const isOpen = button.classList.contains('active');
  
  if (isOpen) {
    content.style.maxHeight = null;
    button.classList.remove('active');
  } else {
    content.style.maxHeight = content.scrollHeight + 'px';
    button.classList.add('active');
  }
}
```

## 📊 Performance Tips

1. **Limit animations**: Use `prefers-reduced-motion` media query
```css
@media (prefers-reduced-motion: reduce) {
  * {
    animation: none !important;
    transition: none !important;
  }
}
```

2. **Debounce resize events**:
```javascript
let resizeTimer;
window.addEventListener('resize', () => {
  clearTimeout(resizeTimer);
  resizeTimer = setTimeout(() => {
    updateLayout();
  }, 200);
});
```

3. **Use CSS transforms** instead of position/width changes:
```javascript
// Good ✓
element.style.transform = 'translateX(100px)';

// Avoid ✗
element.style.left = '100px';
```
